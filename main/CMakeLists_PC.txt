    message("Git found: ${GIT_EXECUTABLE}")

# https://cmake.org/cmake/help/latest/module/FindGTest.html
find_package(GTest 1.11.0 QUIET)
if(GTest_FOUND)
  message("Google Test found")
else()
    message(STATUS "Google Test: not found locally, fetching from release archive...")

    # Once the archive is downloaded and extracted, skip the network check on
    # every subsequent reconfigure. This is the primary cause of the slow
    # cmake -B build . on DrvFs: without this flag CMake re-downloads and
    # re-extracts ~800 kB on every clean configure (300-475 s observed).
    # Delete the build/_deps/googletest-* directories to force a re-fetch.

    set(FETCHCONTENT_UPDATES_DISCONNECTED TRUE)

    # Show curl download progress and extraction steps so the configure does
    # not appear stuck during the first (cold) fetch.
    
    set(FETCHCONTENT_QUIET OFF)

    FetchContent_Declare(
    googletest
    URL      https://github.com/google/googletest/archive/refs/tags/release-1.11.0.tar.gz
    URL_HASH SHA256=b4870bf121ff7795ba20d20bcdd8627b8e088f2d1dab299a031c1034eddc93d5
    DOWNLOAD_EXTRACT_TIMESTAMP TRUE
    )

    if(MSVC OR MSYS OR MINGW)
    set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
    endif()

    set(BUILD_GMOCK OFF CACHE BOOL "" FORCE)
    set(BUILD_GTEST ON CACHE BOOL "" FORCE)

    # Suppress third-party deprecation noise from googletest's own CMake files.
    if(DEFINED CACHE{CMAKE_WARN_DEPRECATED})
        get_property(_saved_cmake_warn_deprecated CACHE CMAKE_WARN_DEPRECATED PROPERTY VALUE)
        set(_had_cmake_warn_deprecated_cache TRUE)
    else()
        set(_saved_cmake_warn_deprecated OFF)
        set(_had_cmake_warn_deprecated_cache FALSE)
    endif()

    set(CMAKE_WARN_DEPRECATED OFF CACHE BOOL "" FORCE)

    FetchContent_MakeAvailable(googletest)

    if(_had_cmake_warn_deprecated_cache)
        set(CMAKE_WARN_DEPRECATED ${_saved_cmake_warn_deprecated} CACHE BOOL "" FORCE)
    else()
        unset(CMAKE_WARN_DEPRECATED CACHE)
    endif()

    unset(_saved_cmake_warn_deprecated)
    unset(_had_cmake_warn_deprecated_cache)
endif()

if(TARGET GTest::gtest_main)
    set(GTEST_MAIN_TARGET GTest::gtest_main)
elseif(TARGET gtest_main)
    set(GTEST_MAIN_TARGET gtest_main)
else()
    message(FATAL_ERROR "GoogleTest main target not available")
endif()

# Create an executable for running tests.
# Do not use `run_tests` on Visual Studio because CTest generates a built-in
# `RUN_TESTS` utility target and Windows project names are case-insensitive.
set(RUN_TESTS_TARGET publish_subscribe_tests)
if(TARGET ${RUN_TESTS_TARGET})
    get_filename_component(_cmake_list_name "${CMAKE_CURRENT_LIST_FILE}" NAME_WE)
    set(RUN_TESTS_TARGET "publish_subscribe_tests_${_cmake_list_name}")
    message(WARNING "Target publish_subscribe_tests already exists. Using ${RUN_TESTS_TARGET} in this CMake list to avoid solution project-name collisions.")
endif()
add_executable(${RUN_TESTS_TARGET} ${TEST_SOURCES})

# Disable clang-tidy for the test target: it runs on every TU and can triple
# per-file compile time. Static analysis of production code is enough.
set_target_properties(${RUN_TESTS_TARGET} PROPERTIES CXX_CLANG_TIDY "")

# Precompiled header: gtest/gtest.h and the STL headers used by most test files
# are parsed once and reused across all 23 TUs instead of being re-parsed each time.
target_precompile_headers(${RUN_TESTS_TARGET} PRIVATE
    <gtest/gtest.h>
    <atomic>
    <chrono>
    <cstddef>
    <cstdint>
    <functional>
    <memory>
    <string>
    <thread>
    <utility>
    <vector>
)

# Link Google Test and threads in a platform-portable way
target_link_libraries(${RUN_TESTS_TARGET} PRIVATE framework_modules project_options ${GTEST_MAIN_TARGET} Threads::Threads)
